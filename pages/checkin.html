function submitCheckin() { const user = auth.currentUser; if (!user) { alert("You must be logged in."); return; } const file = document.getElementById("receipt").files[0]; if (!file) { alert("Please upload a receipt."); return; } const storageRef = storage.ref("receipts/" + user.uid + "/" + file.name); storageRef.put(file) .then(() => alert("Receipt uploaded!")) .catch(error => alert(error.message)); }
